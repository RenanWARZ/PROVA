<body>

  <h1>Mercado - Gerenciamento de Produtos</h1>

  <!-- Formulário de Cadastro -->
  <form id="produtoForm">
    <input type="text" id="nome" placeholder="Nome do Produto" required>
    <input type="number" id="preco" placeholder="Preço (R$)" required>
    <input type="number" id="desconto" placeholder="Desconto (%)" required>
    <button type="submit">Adicionar Produto</button>
  </form>

  <!-- Filtro por nome -->
  <div class="filtro">
    <input type="text" id="filtroNome" placeholder="Buscar Produto pelo Nome">
    <button onclick="listarProduto()">Listar Produto</button>
    <button onclick="atualizarTabela()">Mostrar Todos</button>
  </div>

  <!-- Tabela de Produtos -->
  <table>
    <thead>
      <tr>
        <th>Nome</th>
        <th>Preço</th>
        <th>Desconto</th>
        <th>Preço Final</th>
        <th>Ações</th>
      </tr>
    </thead>
    <tbody id="tabelaProdutos"></tbody>
  </table>

  <script>
    let produtos = [];
    let editIndex = -1;

    const form = document.getElementById('produtoForm');
    const tabela = document.getElementById('tabelaProdutos');

    form.addEventListener('submit', function (e) {
      e.preventDefault();
      const nome = document.getElementById('nome').value;
      const preco = parseFloat(document.getElementById('preco').value);
      const desconto = parseFloat(document.getElementById('desconto').value);

      const precoFinal = preco - (preco * (desconto / 100));
      const produto = { nome, preco, desconto, precoFinal: precoFinal.toFixed(2) };

      if (editIndex === -1) {
        produtos.push(produto);
      } else {
        produtos[editIndex] = produto;
        editIndex = -1;
      }

      form.reset();
      atualizarTabela();
    });

    function atualizarTabela(lista = produtos) {
      tabela.innerHTML = '';
      lista.forEach((prod, index) => {
        const row = `<tr>
          <td>${prod.nome}</td>
          <td>R$ ${prod.preco.toFixed(2)}</td>
          <td>${prod.desconto}%</td>
          <td>R$ ${prod.precoFinal}</td>
          <td class="actions">
            <button onclick="editarProduto(${index})">Editar</button>
            <button onclick="deletarProduto(${index})">Excluir</button>
          </td>
        </tr>`;
        tabela.innerHTML += row;
      });
    }

    function editarProduto(index) {
      const prod = produtos[index];
      document.getElementById('nome').value = prod.nome;
      document.getElementById('preco').value = prod.preco;
      document.getElementById('desconto').value = prod.desconto;
      editIndex = index;
    }

    function deletarProduto(index) {
      produtos.splice(index, 1);
      atualizarTabela();
    }

    function listarProduto() {
      const nomeBusca = document.getElementById('filtroNome').value.toLowerCase();
      const filtrados = produtos.filter(p => p.nome.toLowerCase().includes(nomeBusca));
      atualizarTabela(filtrados);
    }
  </script>
