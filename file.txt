<body>
  <div class="container">

    <h1>Mercadinho Digital</h1>

    <!-- Navegação -->
    <nav>
      <button onclick="showSection('login')">Login</button>
      <button onclick="showSection('catalog')" id="btnCatalog">Catálogo</button>
      <button onclick="showSection('cart')" id="btnCart">Carrinho</button>
      <button onclick="showSection('checkout')" id="btnCheckout">Finalizar</button>
    </nav>

    <!-- Login -->
    <section id="login">
      <h2>Login</h2>
      <form onsubmit="login(event)">
        <input type="text" id="username" placeholder="Digite seu nome" required>
        <button type="submit">Entrar</button>
      </form>
    </section>

    <!-- Catálogo -->
    <section id="catalog" class="hidden">
      <h2>Catálogo de Produtos</h2>
      <div class="products" id="productList">Carregando produtos...</div>
    </section>

    <!-- Carrinho -->
    <section id="cart" class="hidden">
      <h2>Seu Carrinho</h2>
      <div class="cart" id="cartList"></div>
      <p id="totalCart"></p>
    </section>

    <!-- Checkout -->
    <section id="checkout" class="hidden">
      <h2>Finalizar Compra</h2>
      <form onsubmit="finalizarCompra(event)">
        <input type="text" placeholder="Endereço de entrega" required>
        <select required>
          <option value="">Forma de pagamento</option>
          <option>Cartão de Crédito</option>
          <option>PIX</option>
          <option>Boleto</option>
        </select>
        <button type="submit">Confirmar Pedido</button>
      </form>
    </section>
  </div>

  <script>
    // Controle de seções
    function showSection(id) {
      document.querySelectorAll("section").forEach(sec => sec.classList.add("hidden"));
      document.getElementById(id).classList.remove("hidden");
    }

    // Login
    function login(e) {
      e.preventDefault();
      const user = document.getElementById("username").value.trim();
      if (user) {
        localStorage.setItem("mercado_usuario", user);
        loadProducts();
        showSection("catalog");
      }
    }

    // Verifica login automático
    if (localStorage.getItem("mercado_usuario")) {
      loadProducts();
      showSection("catalog");
    }

    // Carregar catálogo
    async function loadProducts() {
      const res = await fetch("https://fakestoreapi.com/products");
      const data = await res.json();
      const container = document.getElementById("productList");
      container.innerHTML = "";
      data.forEach(prod => {
        const div = document.createElement("div");
        div.className = "product";
        div.innerHTML = `
          <img src="${prod.image}">
          <h4>${prod.title}</h4>
          <p>R$ ${prod.price.toFixed(2)}</p>
          <button class="btn" onclick='addToCart(${JSON.stringify(prod)})'>Adicionar ao Carrinho</button>
        `;
        container.appendChild(div);
      });
    }

    // Adicionar ao carrinho
    function addToCart(prod) {
      let cart = JSON.parse(localStorage.getItem("mercado_carrinho")) || [];
      const index = cart.findIndex(p => p.id === prod.id);
      if (index > -1) {
        cart[index].qtd += 1;
      } else {
        cart.push({ ...prod, qtd: 1 });
      }
      localStorage.setItem("mercado_carrinho", JSON.stringify(cart));
      alert("Produto adicionado!");
    }

    // Mostrar carrinho
    document.getElementById("btnCart").addEventListener("click", () => {
      const container = document.getElementById("cartList");
      const totalLabel = document.getElementById("totalCart");
      let cart = JSON.parse(localStorage.getItem("mercado_carrinho")) || [];
      container.innerHTML = "";
      let total = 0;

      cart.forEach((item, i) => {
        const div = document.createElement("div");
        div.className = "cart-item";
        div.innerHTML = `
          <img src="${item.image}">
          <h4>${item.title}</h4>
          <p>Qtd: ${item.qtd}</p>
          <p>Subtotal: R$ ${(item.qtd * item.price).toFixed(2)}</p>
          <button class="btn remove" onclick="removeItem(${i})">Remover</button>
        `;
        container.appendChild(div);
        total += item.qtd * item.price;
      });

      totalLabel.innerText = "Total: R$ " + total.toFixed(2);
    });

    // Remover item do carrinho
    function removeItem(index) {
      let cart = JSON.parse(localStorage.getItem("mercado_carrinho")) || [];
      cart.splice(index, 1);
      localStorage.setItem("mercado_carrinho", JSON.stringify(cart));
      showSection("cart");
      document.getElementById("btnCart").click(); // Recarrega carrinho
    }

    // Finalizar compra
    function finalizarCompra(e) {
      e.preventDefault();
      localStorage.removeItem("mercado_carrinho");
      alert("Compra finalizada com sucesso!");
      showSection("catalog");
    }
  </script>
