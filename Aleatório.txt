<body>
  <h1>CRUD Aleatório de Usuários</h1>

  <button onclick="criarUsuario()">Adicionar Usuário Aleatório</button>

  <ul id="listaUsuarios"></ul>

  <script>
    const nomes = ["Ana", "Carlos", "Beatriz", "João", "Mariana", "Pedro", "Renan", "Luana"];
    let usuarios = [];
    let id = 1;

    function criarUsuario() {
      const nomeAleatorio = nomes[Math.floor(Math.random() * nomes.length)];
      usuarios.push({ id: id++, nome: nomeAleatorio });
      renderizarLista();
    }

    function editarUsuario(id) {
      const novoNome = prompt("Digite o novo nome:");
      if (novoNome) {
        const usuario = usuarios.find(u => u.id === id);
        usuario.nome = novoNome;
        renderizarLista();
      }
    }

    function removerUsuario(id) {
      usuarios = usuarios.filter(u => u.id !== id);
      renderizarLista();
    }

    function renderizarLista() {
      const ul = document.getElementById("listaUsuarios");
      ul.innerHTML = "";

      usuarios.forEach(usuario => {
        const li = document.createElement("li");
        li.innerHTML = `
          <strong>${usuario.nome}</strong>
          <button class="editar" onclick="editarUsuario(${usuario.id})">Editar</button>
          <button class="remover" onclick="removerUsuario(${usuario.id})">Remover</button>
        `;
        ul.appendChild(li);
      });
    }
  </script>
