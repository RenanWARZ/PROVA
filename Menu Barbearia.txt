  <script>
    function mostrarResumo() {
      const checkboxes = document.querySelectorAll('input[name="servico"]:checked');
      const listaResumo = document.getElementById('listaResumo');
      const divResumo = document.getElementById('resumo');

      // Limpa lista anterior
      listaResumo.innerHTML = '';

      if (checkboxes.length === 0) {
        alert("Por favor, selecione pelo menos um serviço.");
        return;
      }

      checkboxes.forEach(cb => {
        const li = document.createElement('li');
        li.textContent = cb.value;
        listaResumo.appendChild(li);
      });

      divResumo.style.display = 'block';
    }
  </script>
