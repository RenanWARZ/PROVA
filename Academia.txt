      <input type="text" id="nome" placeholder="Nome" required>
      <input type="text" id="cpf" placeholder="CPF" required>
      <input type="number" id="idade" placeholder="Idade" required>
      <select id="plano" required>
        <option value="">Plano</option>
        <option value="Mensal">Mensal</option>
        <option value="Trimestral">Trimestral</option>
        <option value="Anual">Anual</option>
      </select>
      <button type="submit" id="btn-submit">Cadastrar</button>

      <thead>
        <tr>
          <th>Nome</th>

          <th>Ações</th>
        </tr>
      </thead>
      <tbody id="tabela-alunos"></tbody>

  <script>
    let alunos = JSON.parse(localStorage.getItem("alunosAcademia")) || [];
    let editIndex = null;

    const form = document.getElementById("form-aluno");
    const nome = document.getElementById("nome");
    const cpf = document.getElementById("cpf");
    const idade = document.getElementById("idade");
    const plano = document.getElementById("plano");
    const btn = document.getElementById("btn-submit");
    const tabela = document.getElementById("tabela-alunos");

    form.onsubmit = function (e) {
      e.preventDefault();
      const aluno = {
        nome: nome.value.trim(),
        cpf: cpf.value.trim(),
        idade: parseInt(idade.value),
        plano: plano.value
      };

      if (editIndex === null) {
        alunos.push(aluno);
      } else {
        alunos[editIndex] = aluno;
        editIndex = null;
        btn.textContent = "Cadastrar";
      }

      salvar();
      form.reset();
      render();
    };

    function render() {
      tabela.innerHTML = "";
      alunos.forEach((aluno, i) => {
        const row = document.createElement("tr");
        row.innerHTML = `
          <td>${aluno.nome}</td>
          <td>${aluno.cpf}</td>
          <td>${aluno.idade}</td>
          <td>${aluno.plano}</td>
          <td>
            <button class="btn btn-edit" onclick="editar(${i})">Editar</button>
            <button class="btn btn-delete" onclick="excluir(${i})">Excluir</button>
          </td>
        `;
        tabela.appendChild(row);
      });
    }

    function editar(i) {
      const aluno = alunos[i];
      nome.value = aluno.nome;
      cpf.value = aluno.cpf;
      idade.value = aluno.idade;
      plano.value = aluno.plano;
      editIndex = i;
      btn.textContent = "Salvar";
    }

    function excluir(i) {
      if (confirm("Deseja remover este aluno?")) {
        alunos.splice(i, 1);
        salvar();
        render();
      }
    }

    function salvar() {
      localStorage.setItem("alunosAcademia", JSON.stringify(alunos));
    }

    render(); // Carrega ao iniciar
  </script>
